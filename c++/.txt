int* prepareChai(int cups){
    int* orders = new int(cups);     // new is the keywords

    for(int i = 0; i < cups; i++){
        orders[i] = (i + 1) * 10;
    }
    return orders;
}

int main(){
    int cups = 5;
    int* orderChai = prepareChai(cups);

    for(int i = 0; i < cups; i++){
        cout << "cups: " << i + 1 << "has" << orderChai[i] << "ml" << endl;
    }
